# MS-SQL

create database

``` sql
CREATE DATABASE database;
```

delete database

``` sql
DROP DATABASE database;
```

switch database

``` sql
USE database;
```

create table

``` sql
CREATE TABLE table (
    column type constraint,
    ...
);
```

delete table

``` sql
DROP TABLE table;
```

insert row

``` sql
INSERT INTO table (column, ...)
VALUES (value, ...);
```

edit row

``` sql
UPDATE table SET column = value;
```

delete rows

``` sql
DELETE FROM table;
```
